\begin{frame}{Funções de Kernel}

\begin{itemize}
\item Técnicas flexíveis para modelagem de relações não lineares;

\vspace{0.3cm}

\item Permitem estender algoritmos lineares para espaços de maior dimensão;

\vspace{0.3cm}

\item Fundamentados em funções de kernel e espaços de Hilbert;

\vspace{0.3cm}

\item Amplamente utilizados em:
\begin{itemize}
    \item Support Vector Machines (SVM);
    \item Kernel Ridge Regression;
    \item Gaussian Processes.
\end{itemize}

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Funções de Kernel}

\begin{itemize}
\item Uma função kernel é definida por:
\end{itemize}

$$
K: \mathcal{X}\times\mathcal{X}\to\mathbb{R}
$$

\vspace{0.3cm}

\begin{itemize}
\item Interpreta similaridade entre observações;

\vspace{0.2cm}

\item Equivale a um produto interno em um espaço de características:

\end{itemize}

$$
K(x,x')=\langle \phi(x),\phi(x')\rangle
$$

\vspace{0.3cm}

\begin{itemize}
\item Evita calcular explicitamente $\phi(x)$.
\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{RKHS}

\begin{itemize}
\item Kernels positivos definidos induzem um espaço de Hilbert;

\vspace{0.3cm}

\item A matriz de Gram deve satisfazer:

\end{itemize}

$$
K=[K(x_i,x_j)]_{i,j=1}^n \succeq 0
$$

\vspace{0.3cm}

\begin{itemize}
\item O espaço associado a K é chamado de Reproducing Kernel Hilbert Space (RKHS);
\end{itemize}


\end{frame}

% %--------------------------------------------------------
\begin{frame}{Teorema do Representante}


Seja $K\text{ : } \mathcal{X} \times \mathcal{X} \to \mathbb{R}$ um kernel PDS e $\mathcal{H}$
seu RHKS correspondente. Então, para qualquer função não-descrescente $G\text{ : } \mathbb{R} \to \mathbb{R}$
e qualquer função de perda $\mathcal{L}\text{ : } \mathbb{R}^n \to \mathbb{R} \cup \{+\infty\}$, a solução do problema

\[
\arg\min_{h\in \mathcal{H}} \left\{ \mathcal{L}(h(x_1), \dots, h(x_n)) + G(\|h\|_{\mathcal{H}}) \right\}
% {\arg\min}_{h \in \mathcal{H}} \left\{ \mathcal{L}(h(x_1), \dots, h(x_n)) + \lambda \|h\|_{\mathcal{H}}^2 \right\}
\]

% \[
% \min_{f \in \mathcal{H}}
% \left\{
% \mathcal{L}(f(x_1), \dots, f(x_n))
% +
% \lambda \|f\|_{\mathcal{H}}^2
% \right\}
% \]

admite uma solução na forma

\[
h^*(x)=\sum_{i=1}^{n}\alpha_i K(x_i,x)\text{.}
\]

\end{frame}

\section{Modelos de Longa Duração}
\setLayout{vertical}

%--------------------------------------------------------
\begin{frame}{Modelos de Longa Duração}

\begin{itemize}
\item Em alguns estudos, parte dos indivíduos nunca experimenta o evento de interesse;

\vspace{0.3cm}

\item Esses indivíduos são chamados de:
\begin{itemize}
    \item imunes;
    \item curados.
\end{itemize}

\vspace{0.3cm}

\item Exemplos:
\begin{itemize}
    \item recorrência de câncer;
    \item falha de dispositivos;
    \item inadimplência.
\end{itemize}

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Fração de Cura}

\begin{itemize}
\item Modelos clássicos assumem:
\end{itemize}

$$
\lim_{t\to\infty} S(t)=0
$$

\vspace{0.4cm}

\begin{itemize}
\item Em modelos de longa duração:
\end{itemize}

$$
\lim_{t\to\infty} S(t)=p_0>0
$$

\vspace{0.4cm}

\begin{itemize}
\item $p_0$ representa a fração de cura da população.
\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Função de Sobrevivência Imprópria}

\begin{itemize}
\item A função de sobrevivência continua definida como:
\end{itemize}

$$
S(t)=P(T>t)
$$

\vspace{0.4cm}

\begin{itemize}
\item Mas agora admite:
\end{itemize}

$$
P(T=\infty)=p_0
$$

\vspace{0.4cm}

\begin{itemize}
\item Isso gera uma sobrevivência imprópria;

\vspace{0.2cm}

\item A curva estabiliza em um patamar positivo.

\end{itemize}

\end{frame}

%========================================================
\section{Modelo de Tempo de Promoção}

%--------------------------------------------------------
\begin{frame}{Modelo de Tempo de Promoção}

\begin{itemize}
\item Também conhecido como:
\begin{itemize}
\item \textit{Promotion Time Cure Model};
\item \textit{Bounded Cumulative Hazard Model}.
\end{itemize}

\vspace{0.4cm}

\item Baseado na ideia de riscos latentes;

\vspace{0.3cm}

\item Exemplo: células cancerígenas que podem promover a recorrência da doença.

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Formulação do Modelo}

\begin{itemize}
\item Seja $N$ o número de riscos latentes:
\end{itemize}

$$
N \sim Poisson(\theta(x))
$$

\vspace{0.4cm}

\begin{itemize}
\item Cada risco possui tempo de promoção:
\end{itemize}

$$
T_1,T_2,\ldots,T_N \sim F(t)
$$

\vspace{0.4cm}

\begin{itemize}
\item O tempo observado é dado pelo mínimo:
\end{itemize}

$$
T=\min(T_1,\ldots,T_N)
$$

\end{frame}

% %--------------------------------------------------------
\begin{frame}{Sobrevivência Populacional}

\begin{itemize}
\item A função de sobrevivência populacional é:
\end{itemize}

$$
S_{pop}(t|x)=P(T>t)
$$

\vspace{0.3cm}

\begin{itemize}
\item Sob independência:
\end{itemize}

$$
S_{pop}(t|x)=\exp\{-\theta(x)F(t)\}
$$

\vspace{0.4cm}

\begin{itemize}
\item A fração de cura é obtida por:
\end{itemize}

$$
p_0(x)=\lim_{t\to\infty}S_{pop}(t|x)=\exp(-\theta(x))
$$

\end{frame}

\section{GG-KM}

\begin{frame}
\frametitle{Extensão Não Paramétrica via Kernel}
\end{frame}

\setLayout{vertical}

%--------------------------------------------------------
\begin{frame}{Verossimilhança Penalizada}

\begin{itemize}
\item Seja $(x_i,t_i,\delta_i)$ o conjunto de dados observados;

\vspace{0.3cm}

\item Considerando:

\end{itemize}

$$
\theta(x)=\exp(f(x))
$$

\vspace{0.3cm}

\begin{itemize}
\item A função objetivo penalizada é dada por:
\end{itemize}

$$
J(\Theta)=
\left[
\delta^\top (K\alpha+\log f_{GG})
w^\top F_{GG}
\right]
+
\frac{\lambda}{2}\alpha^\top K\alpha
$$

\vspace{0.3cm}

\begin{itemize}
\item onde:
\end{itemize}

$$
w=\exp(K\alpha)
$$

\end{frame}

% %--------------------------------------------------------
\begin{frame}{Justificativa via Teorema do Representante}

\begin{itemize}
\item Supondo que $f(x)$ pertença a um RKHS;

\vspace{0.3cm}

\item Pelo Teorema do Representante, a solução ótima admite:

\end{itemize}

$$
f(x)=\sum_{j=1}^{n}\alpha_jK(x,x_j)
$$

\vspace{0.4cm}

\begin{itemize}
\item Isso reduz um problema infinito-dimensional para dimensão finita;

\vspace{0.3cm}

\item Assim, toda a estimação depende apenas da matriz kernel.

\end{itemize}

\end{frame}

\begin{frame}[fragile]
\frametitle{Algoritmo de Estimação do GG-KM}

\begin{algorithm}[H]
\caption{Processo de estimação do modelo GG-KM}
\begin{algorithmic}[1]
\State \textbf{Entrada:} Dados $(X,t,\delta)$, hiperparâmetros $(\lambda_{reg},\gamma)$
\State Inicializar $\alpha \leftarrow 0$ e $\theta_{GG} \leftarrow {a=1,d=1,p=1}$
\State Construir a matriz de Gram $K$, com $K_{ij}=K(\mathbf{x}_i,\mathbf{x}_j)$

\While{critério de convergência não satisfeito}
\State Calcular $w_i=\exp\left(\sum_j\alpha_jK_{ij}\right)$
\State Calcular $f_{GG}(t_i)$ e $F_{GG}(t_i)$
\State Avaliar $J(\alpha,\theta_{GG})$
\State Calcular $\nabla_{\alpha}J$ e $\nabla_{\theta_{GG}}J$
\State Atualizar $\Theta={\alpha,a,d,p}$ via L-BFGS
\EndWhile

\State \textbf{Saída:} $\hat{\alpha},\hat{a},\hat{d},\hat{p}$
\end{algorithmic}
\end{algorithm}

\end{frame}

\section{Geração dos Dados Simulados}

%--------------------------------------------------------
\setLayout{vertical}
\begin{frame}{Estrutura dos Dados Simulados}

\begin{itemize}
\item Foram geradas duas covariáveis independentes:
\end{itemize}

\[
x_1,x_2 \sim N(0,1)
\]

\vspace{0.4cm}

\begin{itemize}
\item A estrutura de cura foi construída a partir de três cenários:

\begin{itemize}
    \item linear;
    \item quadrático;
    \item não linear periódico.
\end{itemize}

\vspace{0.3cm}

\item Objetivo:
\begin{itemize}
    % \item avaliar robustez;
    \item avaliar flexibilidade do GG-KM.
\end{itemize}

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Cenários para Fração de Cura}

\textbf{Cenário 1: Linear}

\[
\pi(x)=
\frac{\exp(0.3-5x_1-3x_2)}
{1+\exp(0.3-5x_1-3x_2)}
\]

\vspace{0.3cm}

\textbf{Cenário 2: Quadrático}

\[
\pi(x)=
\frac{\exp(0.3+5x_1^2-3x_2^2)}
{1+\exp(0.3+5x_1^2-3x_2^2)}
\]

\vspace{0.3cm}

\textbf{Cenário 3: Não linear periódico}

\[
\pi(x)=
\exp\{-\exp(0.3-5\cos(x_1)-3\sin(x_2))\}
\]

\end{frame}

%--------------------------------------------------------
\begin{frame}{Geração dos Dados Simulados: Passo 1}

\begin{itemize}
\item Para cada indivíduo $i=1,\ldots,n$:

\vspace{0.4cm}

\item Gerar:

\end{itemize}

\[
U_i \sim Uniform(0,1)
\]

\[
C_i \sim Exp(0.2)
\]

\vspace{0.4cm}

\begin{itemize}
\item $U_i$: determina se o indivíduo será suscetível ou curado;

\vspace{0.2cm}

\item $C_i$: tempo de censura.

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Geração dos Dados Simulados: Passo 2}

\begin{itemize}
\item Verificar:
\end{itemize}

\[
U_i \leq 1-\pi(x_i)
\]

\vspace{0.5cm}

Se verdadeiro, indivíduo pertence ao grupo curado, e portanto:

\vspace{0.3cm}

\[
t_i=C_i
\]

\[
\delta_i=0
\]

\end{frame}

%--------------------------------------------------------
\begin{frame}{Geração dos Dados Simulados: Passo 3}

\begin{itemize}
\item Caso contrário, o indivíduo é suscetível;

\vspace{0.3cm}

\item Gerar:

\end{itemize}

\[
U_i^* \sim Uniform(1-\pi(x_i),1)
\]

\vspace{0.4cm}

\begin{itemize}
\item Obter o tempo latente:
\end{itemize}

\[
Y_i=
F^{-1}
\left(
\frac{\log(U_i^*)}
{\log(1-\pi(x_i))}
\right)
\]

\end{frame}

%--------------------------------------------------------
\begin{frame}{Geração dos Dados Simulados: Passo 4}

\begin{itemize}
\item Comparar tempo latente e censura:
\end{itemize}

\[
t_i=\min(Y_i,C_i)
\]

\vspace{0.4cm}

\begin{itemize}
\item Indicador do evento:
\end{itemize}

\[
\delta_i=
\begin{cases}
1, & t_i=Y_i \\
0, & t_i=C_i
\end{cases}
\]

\vspace{0.4cm}

Por fim,  obtém-se $(x_i,t_i,\delta_i)$ para cada indivíduo.
\end{frame}

\begin{frame}{Métricas para avaliação do desempenho do modelo}

Integrated Brier Score com ajuste IPCW:

\[
IBS=
\frac{1}{\tau}
\int_0^\tau BS_{IPCW}(t)\,dt,
\]

\[
BS_{IPCW}(t)
=
\frac{1}{\tilde{n}(t)}
\sum_{i=1}^{n}
\left[
\hat{S}_i(t)^2
\frac{\mathbb{I}(T_i^*\leq t,\delta_i=1)}{\hat{G}(T_i^*)}
+
(1-\hat{S}_i(t))^2
\frac{\mathbb{I}(T_i^*>t)}{\hat{G}(t)}
\right],
\]
em que
\[
\tilde{n}(t)=\sum_{i=1}^{n}\omega_i(t)\text{,  } \omega_i(t)= \frac{\mathbb{I}(T_i^*\leq t,\delta_i=1)}{\hat{G}(T_i^*)} + \frac{\mathbb{I}(T_i^*>t)}{\hat{G}(t)}.
\]

\end{frame}

\begin{frame}{Métricas para avaliação do desempenho do modelo}
$$
\text{Viés}\left(\widehat{\pi}\left(\mathbf{x}\right)\right) =
\frac{1}{N_1} \sum_{r=1}^{N_1} \left[\frac{1}{n}\sum_{i=1}^{n}\{\hat{\pi^{\left(r\right)}}\left(\mathbf{x}_i\right) -
\pi^{\left(r\right)}\left(\mathbf{x}_i\right)\}\right]\text{,}
$$

$$
\text{EQM}\left(\widehat{\pi}\left(\mathbf{x}\right)\right) = \frac{1}{N_1} \sum_{r=1}^{N_1} \left[\frac{1}{n}\sum_{i=1}^{n}\left\{\hat{\pi^{\left(r\right)}}\left(\mathbf{x}_i\right) - \pi^{\left(r\right)}\left(\mathbf{x}_i\right)\right\}^2\right]\text{.}
$$
\end{frame}
