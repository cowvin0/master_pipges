\begin{frame}{SVM: Formulação Primal}

\textbf{Problema de otimização:}

\vspace{0.3cm}

\begin{equation*}
\min_{w,b,\boldsymbol{\xi}} \;\; \frac{1}{2}\|w\|^2 + C \sum_{i=1}^{m} \xi_i
\end{equation*}

\vspace{0.3cm}

\textbf{Sujeito a:}

\begin{equation*}
y_i (w^\top x_i + b) \geq 1 - \xi_i, \quad i = 1, \dots, m
\end{equation*}

\begin{equation*}
\xi_i \geq 0, \quad i = 1, \dots, m
\end{equation*}

\vspace{0.3cm}

\begin{itemize}
    \item $C$: parâmetro de regularização
    \item $\xi_i$: variáveis de folga
\end{itemize}

\end{frame}

\begin{frame}{SVM: Formulação Dual}

\textbf{Problema dual (Lagrangiano):}

\vspace{0.3cm}

\begin{equation*}
\max_{\boldsymbol{\alpha}} \;\; \sum_{i=1}^{m} \alpha_i
- \frac{1}{2} \sum_{i=1}^{m} \sum_{j=1}^{m} \alpha_i \alpha_j y_i y_j \langle x_i, x_j \rangle
\end{equation*}

\vspace{0.3cm}

\textbf{Sujeito a:}

\begin{equation*}
0 \leq \alpha_i \leq C, \quad i = 1, \dots, m
\end{equation*}

\begin{equation*}
\sum_{i=1}^{m} \alpha_i y_i = 0
\end{equation*}

\vspace{0.3cm}

\begin{itemize}
    \item $\alpha_i$: multiplicadores de Lagrange
    % \item Base para uso de kernels
\end{itemize}
\end{frame}

\begin{frame}{Problema Primal (SVR)}

\begin{equation*}
\min_{w,b,\xi,\xi'} \;\;
\frac{1}{2} \|w\|^2 + C \sum_{i=1}^{m} (\xi_i + \xi_i')
\end{equation*}

\vspace{0.3cm}

\textbf{sujeito a:}

\begin{equation*}
w \cdot \Phi(x_i) + b - y_i \leq \epsilon + \xi_i
\end{equation*}

\begin{equation*}
y_i - w \cdot \Phi(x_i) - b \leq \epsilon + \xi_i'
\end{equation*}

\begin{equation*}
\xi_i, \xi_i' \geq 0
\end{equation*}

\end{frame}

\begin{frame}{Problema Dual (SVR)}

\begin{equation*}
\max_{\alpha,\alpha'} \;\;
\epsilon \sum (\alpha_i + \alpha_i')
- \sum y_i (\alpha_i' - \alpha_i)
\end{equation*}

\begin{equation*}
- \frac{1}{2} \sum_{i,j} (\alpha_i' - \alpha_i)(\alpha_j' - \alpha_j) K_{ij}
\end{equation*}

\vspace{0.3cm}

\textbf{sujeito a:}

\begin{equation*}
\sum (\alpha_i' - \alpha_i) = 0
\end{equation*}

\begin{equation*}
0 \leq \alpha_i, \alpha_i' \leq C
\end{equation*}

\end{frame}

\begin{frame}{Solução Final}

\begin{equation*}
h(x) = \sum_{i=1}^{m} (\alpha_i' - \alpha_i) K(x_i, x) + b
\end{equation*}

\vspace{0.4cm}

\begin{itemize}
    \item Apenas vetores de suporte contribuem
    \item Pontos dentro do tubo:
    \begin{itemize}
        \item $\alpha_i = \alpha_i' = 0$
    \end{itemize}
\end{itemize}

\end{frame}

\section{Support Vector Censored Regression}

\begin{frame}{Ideia central}
\begin{itemize}
    \item Extensão do SVR para dados com censura
\end{itemize}

\vspace{0.4cm}

\begin{itemize}
    \item Dados: $(\mathbf{x}_i, l_i, u_i)$
    \item Tempo verdadeiro: $T_i \in [l_i, u_i]$
\end{itemize}

\vspace{0.4cm}

\begin{itemize}
    \item Sem censura: $l_i = u_i$
    \item Censura à direita: $u_i = +\infty$
    \item Censura à esquerda: $l_i = -\infty$
\end{itemize}

\end{frame}

% \end{itemize}
% \end{frame}

\begin{frame}{Função de perda}
\begin{itemize}
    \item Sem penalização dentro do intervalo
    \item Penaliza apenas violações
\end{itemize}

\vspace{0.3cm}

\begin{equation*}
c(f(x_i), l_i, u_i) = \max\{l_i - f(x_i),\, f(x_i) - u_i,\, 0\}
\end{equation*}

\vspace{0.3cm}

% \begin{itemize}
%     \item Generaliza censura à direita, esquerda e intervalar
%     \item Função convexa
% \end{itemize}
\end{frame}

\begin{frame}{Formulação primal}
\begin{equation*}
\min_{w,b,\xi,\xi^*} \frac{1}{2}\|w\|^2 + C \left( \sum_{i\in U}\xi_i + \sum_{i\in L}\xi_i^* \right)
\end{equation*}

\begin{itemize}
    \item Restrições:
    \begin{itemize}
        \item $w^\top x_i + b \leq u_i + \xi_i^*$
        \item $w^\top x_i + b \geq l_i - \xi_i$
    \end{itemize}
    \item $\xi_i, \xi_i^* \geq 0$
\end{itemize}
\end{frame}

\begin{frame}{Formulação dual}
\begin{equation*}
\max_{\alpha,\alpha^*}
- \frac{1}{2} \sum_{i,j} (\alpha_i - \alpha_i^*)(\alpha_j - \alpha_j^*) K(x_i,x_j)
- \sum_{i\in L} l_i \alpha_i + \sum_{i\in U} u_i \alpha_i^*
\end{equation*}

\begin{itemize}
    \item $0 \leq \alpha_i, \alpha_i^* \leq C$
    \item $\sum \alpha_i = \sum \alpha_i^*$
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
    \item Problema quadrático convexo
\end{itemize}
\end{frame}

\begin{frame}{Funções Wavelet}
\begin{itemize}
    \item Ferramenta para \textbf{decompor funções e sinais}
    \item Capturam simultaneamente:
    \begin{itemize}
        \item \textbf{localização} (onde ocorre)
        \item \textbf{escala} (nível de detalhe)
    \end{itemize}
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
    \item Diferente de bases globais (ex.: Fourier)
    \item Permitem análise \textbf{local} e \textbf{multiescala}
\end{itemize}

\vspace{0.3cm}

\begin{equation*}
h_{A,B}(x) = |A|^{-1/2} \, h\left(\frac{x - B}{A}\right)
\end{equation*}

\vspace{0.3cm}

\begin{itemize}
    \item $A$: escala → controla o nível de detalhe
    \item $B$: translação → desloca a função no domínio
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
    \item Aproximação de funções:
\end{itemize}

\begin{equation*}
\hat{g}(x) = \sum_{i=1}^{t} w_i \, h_{A_i,B_i}(x)
\end{equation*}

\begin{itemize}
    \item Ideal para fenômenos \textbf{não lineares} e \textbf{não estacionários}
\end{itemize}
\end{frame}

\begin{frame}{Kernel Wavelet}

\vspace{0.3cm}

\begin{itemize}
    \item Wavelet mãe:
\end{itemize}

\begin{equation*}
h(x) = \cos(1.75x)\exp(-0.5x^2)
\end{equation*}

\vspace{0.3cm}

\begin{itemize}
    \item Kernel multidimensional:
\end{itemize}

\begin{equation*}
K(\mathbf{x}, \mathbf{x}') =
\prod_{k=1}^{p} h\left(\frac{x_k - x_k'}{A}\right)
\end{equation*}

\vspace{0.3cm}

\begin{itemize}
    \item Compara diferenças \textbf{localmente} em cada dimensão
    \item $A$: controla a escala da similaridade
\end{itemize}

\vspace{0.3cm}

\end{frame}

\begin{frame}{Kernels considerados}

\scriptsize

\textbf{Linear}
\[
K(\mathbf{x}, \mathbf{x}') = \mathbf{x}^\top \mathbf{x}'
\]

\textbf{Polinomial}
\[
K(\mathbf{x}, \mathbf{x}') = (\gamma \, \mathbf{x}^\top \mathbf{x}' + r)^d
\]

\textbf{Gaussiano}
\[
K(\mathbf{x}, \mathbf{x}') = \exp\left(-\gamma \|\mathbf{x} - \mathbf{x}'\|^2 \right)
\]

\textbf{Exponencial}
\[
K(\mathbf{x}, \mathbf{x}') = \exp\left(-\gamma \|\mathbf{x} - \mathbf{x}'\| \right)
\]

\textbf{Laplaciano}
\[
K(\mathbf{x}, \mathbf{x}') = \exp\left(-\gamma \|\mathbf{x} - \mathbf{x}'\|_1 \right)
\]

\textbf{Cauchy}
\[
K(\mathbf{x}, \mathbf{x}') = \frac{1}{1 + \gamma \|\mathbf{x} - \mathbf{x}'\|^2}
\]

\textbf{Wavelet}
\[
K(\mathbf{x}, \mathbf{x}') = \prod_{k=1}^{p}
\cos\left(1.75 \frac{x_k - x_k'}{A}\right)
\exp\left(-\frac{1}{2} \left(\frac{x_k - x_k'}{A}\right)^2 \right)
\]

\end{frame}

\begin{frame}{Avaliação dos modelos}
\begin{itemize}
    \item Métrica principal: C-index (concordance index)
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
    \item Mede concordância entre risco predito $v(\mathbf{x})$ e tempo observado
\end{itemize}

\vspace{0.3cm}

\begin{equation*}
C =
\frac{
\sum_{i \neq j} \mathbb{I}\Big[(v(\mathbf{x}_i) - v(\mathbf{x}_j))(y_i - y_j) \geq 0\Big]
}{
\sum_{i \neq j} \text{comp}(i,j)
}
\end{equation*}

\vspace{0.2cm}

\begin{itemize}
    \item $\text{comp}(i,j)$: pares comparáveis (considera censura)
    \item $C = 1$: concordância perfeita
\end{itemize}

\vspace{0.3cm}

\begin{itemize}
    \item Resultados agregados: média e desvio padrão
\end{itemize}

\end{frame}
