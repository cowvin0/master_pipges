\begin{frame}{Representer theorem}

\vfill

% \begin{teo}

Seja $K\text{ : } \mathcal{X} \times \mathcal{X} \to \mathbb{R}$ um kernel PDS e $\mathcal{H}$
seu RHKS correspondente. Então, para qualquer função não-descrescente $G\text{ : } \mathbb{R} \to \mathbb{R}$
e qualquer função de perda $\mathcal{L}\text{ : } \mathbb{R}^n \to \mathbb{R} \cup \{+\infty\}$, a solução do problema

\[
{\arg\min}_{f \in \mathcal{H}} \left\{ \mathcal{L}(f(x_1), \dots, f(x_n)) + \lambda \|f\|_{\mathcal{H}}^2 \right\}
\]

% \[
% \min_{f \in \mathcal{H}}
% \left\{
% \mathcal{L}(f(x_1), \dots, f(x_n))
% +
% \lambda \|f\|_{\mathcal{H}}^2
% \right\}
% \]

admite uma solução na forma

\[
f^*(x)=\sum_{i=1}^{n}\alpha_i K(x_i,x)\text{.}
\]

% \end{teo}

\end{frame}

\begin{frame}{SVM e Representer Theorem}

No SVM, resolvemos o seguinte problema de otimização:

\[
\begin{aligned}
\min_{\mathbf{w},\, b} \quad & \frac{1}{2}\|\mathbf{w}\|^2 \\
\text{sujeito a} \quad & y_i\bigl(\mathbf{w}^\top x_i + b\bigr) \geq 1,
\quad i \in [n]
\end{aligned}
\]

A solução ótima $\mathbf{w}^*$ pode ser expressa como uma combinação linear, como visto no Representer Theorem:

\[
\mathbf{w}^* = \sum_{i=1}^{n} \alpha_i y_i x_i\text{, em que } y_i \in \{-1, 1\}\text{.}
\]

Portanto, o problema se torna equivalente a otimizar sobre os coeficientes $\alpha_1, \dots, \alpha_n$ em vez de $\mathbf{w}$.

\end{frame}

\begin{frame}{SVM e Representer Theorem}

Dessa forma, o problema de otimização do SVM pode ser reformulado como:

\[
\begin{aligned}
\min \quad & \frac{1}{2}\sum_{i}\sum_{j}\alpha_i\alpha_j y_i y_j \left\langle x_i, x_j\right\rangle \\
\text{sujeito a} \quad & y_i\left( \sum_{j}\alpha_j y_j \left\langle x_j, x_i\right\rangle + b \right) \geq 1, \quad i \in [n]
\end{aligned}
\]


\end{frame}

\begin{frame}{Kernel Ridge Regression e Representer Theorem}

No Kernel Ridge Regression, resolvemos o seguinte problema de otimização:

\[
\begin{aligned}
\min_{\mathbf{w}} \quad & \sum_{i=1}^n \left(\mathbf{w}\Phi\left(x_i\right) - y_i\right)^2 \\
\text{sujeito a} \quad & \|\mathbf{w}\|^2 \leq \Lambda^2
\end{aligned}
\]

Pelas condições de KKT, $\mathbf{w}$ pode ser determinado por
$$
\mathbf{w} = \sum_{i=1}^n\alpha_i\Phi\left(x_i\right) = \mathbf{X}\mathbf{\alpha}\text{, } \mathbf{\alpha} = \mathbf{X}\left(\mathbf{K} + \lambda \mathbf{I}\right)^{-1}\mathbf{Y}\text{,}
$$
em que a resposta final é dada por $h\left(x\right) = \mathbf{w}\Phi\left(x\right) = \sum_{i=1}^n \alpha_i K\left(x_i, x\right)$.

\end{frame}
