\begin{frame}{Um pouco de notação matemática (horizontal)}
    \textbf{Exemplo básico:}
    $$
    H_0: \mathbb{E}(X) = 0 \Longrightarrow Pg(H_0): |\mathbb{E}(X)| < \varepsilon.
    $$
    \pause
    \begin{itemize}
        \item $\varepsilon$ é o limiar;
        \item $d(\cdot, \cdot)$ é a função de dissimilaridade;
        \item $\mathbb{H}$ é o epaço de hipóteses.
    \end{itemize}
    \pause{}
    \vspace{.1in}
    \begin{equation*}
        Pg(H_0) = \bigcup_{h \in H_0}\{h^* \in \mathbb{H}: d(h,h^*) < \varepsilon\} = \left\{h^* \in \mathbb{H}: \inf_{h \in H_0} d(h,h^*) <\varepsilon \right\}.
    \end{equation*}
\end{frame}
