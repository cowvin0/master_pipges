%--------------------------------------------------------
\setLayout{vertical}
\begin{frame}{Funções de Kernel}

\begin{itemize}
\item Técnicas flexíveis para modelagem de relações não lineares;

\vspace{0.3cm}

\item Permitem estender algoritmos lineares para espaços de maior dimensão;

\vspace{0.3cm}

\item Fundamentados em funções de kernel e espaços de Hilbert;

\vspace{0.3cm}

\item Amplamente utilizados em:
\begin{itemize}
    \item Support Vector Machines (SVM);
    \item Kernel Ridge Regression;
    \item Gaussian Processes.
\end{itemize}

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Funções de Kernel}

\begin{itemize}
\item Uma função kernel é definida por:
\end{itemize}

$$
K: \mathcal{X}\times\mathcal{X}\to\mathbb{R}
$$

\vspace{0.3cm}

\begin{itemize}
\item Interpreta similaridade entre observações;

\vspace{0.2cm}

\item Equivale a um produto interno em um espaço de características:

\end{itemize}

$$
K(x,x')=\langle \phi(x),\phi(x')\rangle
$$

\vspace{0.3cm}

\begin{itemize}
\item Evita calcular explicitamente $\phi(x)$.
\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{RKHS}

\begin{itemize}
\item Kernels positivos definidos induzem um espaço de Hilbert;

\vspace{0.3cm}

\item A matriz de Gram deve satisfazer:

\end{itemize}

$$
K=[K(x_i,x_j)]_{i,j=1}^n \succeq 0
$$

\vspace{0.3cm}

\begin{itemize}
\item O espaço associado a K é chamado de Reproducing Kernel Hilbert Space (RKHS);
\end{itemize}


\end{frame}

% %--------------------------------------------------------
\begin{frame}{Teorema do Representante}

\begin{itemize}
\item Fundamental para problemas de estimação com kernels;

\vspace{0.3cm}

\item Garante que a solução ótima possui representação finita:

\end{itemize}

$$
h^*(x)=\sum_{i=1}^n \alpha_i K(x_i,x)
$$

\vspace{0.4cm}

\begin{itemize}
\item Reduz um problema infinito-dimensional para um problema finito;

\vspace{0.3cm}

\item Toda a informação necessária está contida na matriz kernel.

\end{itemize}

\end{frame}

\section{Modelos de Longa Duração}
\setLayout{vertical}

%--------------------------------------------------------
\begin{frame}{Modelos de Longa Duração}

\begin{itemize}
\item Em alguns estudos, parte dos indivíduos nunca experimenta o evento de interesse;

\vspace{0.3cm}

\item Esses indivíduos são chamados de:
\begin{itemize}
    \item imunes;
    \item curados.
\end{itemize}

\vspace{0.3cm}

\item Exemplos:
\begin{itemize}
    \item recorrência de câncer;
    \item falha de dispositivos;
    \item inadimplência.
\end{itemize}

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Fração de Cura}

\begin{itemize}
\item Modelos clássicos assumem:
\end{itemize}

$$
\lim_{t\to\infty} S(t)=0
$$

\vspace{0.4cm}

\begin{itemize}
\item Em modelos de longa duração:
\end{itemize}

$$
\lim_{t\to\infty} S(t)=p_0>0
$$

\vspace{0.4cm}

\begin{itemize}
\item $p_0$ representa a fração de cura da população.
\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Função de Sobrevivência Imprópria}

\begin{itemize}
\item A função de sobrevivência continua definida como:
\end{itemize}

$$
S(t)=P(T>t)
$$

\vspace{0.4cm}

\begin{itemize}
\item Mas agora admite:
\end{itemize}

$$
P(T=\infty)=p_0
$$

\vspace{0.4cm}

\begin{itemize}
\item Isso gera uma sobrevivência imprópria;

\vspace{0.2cm}

\item A curva estabiliza em um patamar positivo.

\end{itemize}

\end{frame}

%========================================================
\section{Modelo de Tempo de Promoção}

%--------------------------------------------------------
\begin{frame}{Modelo de Tempo de Promoção}

\begin{itemize}
\item Também conhecido como:
\begin{itemize}
\item \textit{Promotion Time Cure Model};
\item \textit{Bounded Cumulative Hazard Model}.
\end{itemize}

\vspace{0.4cm}

\item Baseado na ideia de riscos latentes;

\vspace{0.3cm}

\item Exemplo: células cancerígenas que podem promover a recorrência da doença.

\end{itemize}

\end{frame}

%--------------------------------------------------------
\begin{frame}{Formulação do Modelo}

\begin{itemize}
\item Seja $N$ o número de riscos latentes:
\end{itemize}

$$
N \sim Poisson(\theta(x))
$$

\vspace{0.4cm}

\begin{itemize}
\item Cada risco possui tempo de promoção:
\end{itemize}

$$
T_1,T_2,\ldots,T_N \sim F(t)
$$

\vspace{0.4cm}

\begin{itemize}
\item O tempo observado é dado pelo mínimo:
\end{itemize}

$$
T=\min(T_1,\ldots,T_N)
$$

\end{frame}

% %--------------------------------------------------------
\begin{frame}{Sobrevivência Populacional}

\begin{itemize}
\item A função de sobrevivência populacional é:
\end{itemize}

$$
S_{pop}(t|x)=P(T>t)
$$

\vspace{0.3cm}

\begin{itemize}
\item Sob independência:
\end{itemize}

$$
S_{pop}(t|x)=\exp\{-\theta(x)F(t)\}
$$

\vspace{0.4cm}

\begin{itemize}
\item A fração de cura é obtida por:
\end{itemize}

$$
p_0(x)=\lim_{t\to\infty}S_{pop}(t|x)=\exp(-\theta(x))
$$

\end{frame}
